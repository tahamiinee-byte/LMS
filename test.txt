  <script>
     async function loadUser() {
      try {
        const res = await fetch('/api/users');
        if (!res.ok) throw new Error('Network response was not ok');
        const data = await res.json();

        const user = Array.isArray(data) ? data[0] : data;

        // Use firstname and lastname
        const firstname = user?.firstname || '';
        const lastname = user?.lastname || '';
        const name = `${firstname} ${lastname}`.trim() || 'Student';

        // Set greeting
        const nameEl = document.getElementById('userName');
        nameEl.textContent = name;
        nameEl.classList.remove('loading');

        // Avatar initials
        const initials = name.split(' ').map(w => w[0]).join('').slice(0, 2).toUpperCase();
        document.getElementById('avatarEl').textContent = initials;

      } catch (err) {
        const nameEl = document.getElementById('userName');
        nameEl.classList.remove('loading');
        nameEl.innerHTML = '<span class="error-badge"><svg width="10" height="10" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><circle cx="12" cy="12" r="10"/><line x1="12" y1="8" x2="12" y2="12"/><line x1="12" y1="16" x2="12.01" y2="16"/></svg>Unavailable</span>';
        document.getElementById('avatarEl').textContent = '?';
        console.warn('Could not load user:', err);
      }
    }
